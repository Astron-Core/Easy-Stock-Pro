const GAS_API_URL = "https://script.google.com/macros/s/AKfycbxV4vTpMUR_J6Wx48MtO_90xpXybXaA5pyd6t1daxC3hxG_qXue8hHbhWiK8WYHIPeU/exec"; // เอา URL จากการ Deploy Web App มาใส่ function doLogin() { const usr = document.getElementById('login-usr').value; const pwd = document.getElementById('login-pwd').value; if(!usr || !pwd) return Swal.fire({ icon: 'warning', title: 'Oops!', text: 'Please enter Username & Password' }); Swal.fire({ title: 'Logging in...', allowOutsideClick: false, didOpen: () => { Swal.showLoading(); } }); // สร้างข้อมูลที่จะส่งไป const requestBody = { action: "checkLogin", payload: { username: usr, password: pwd } }; // ยิง Request แบบ POST ไปที่ Google Apps Script fetch(GAS_API_URL, { method: 'POST', body: JSON.stringify(requestBody), headers: { 'Content-Type': 'text/plain;charset=utf-8', // GAS แนะนำให้ใช้ text/plain เพื่อเลี่ยงปัญหา CORS บางประการ } }) .then(response => response.json()) .then(res => { if(res.success) { currentUser = res.user; localStorage.setItem('es_user_id', currentUser.id); Swal.fire({ icon: 'success', title: 'Welcome!', text: 'Hello ' + currentUser.givenName, timer: 1500, showConfirmButton: false }); setupAppUI(); } else { Swal.fire({ icon: 'error', title: 'Login Failed', text: res.message }); } }) .catch(err => { Swal.fire({ icon: 'error', title: 'System Error', text: 'Error in checkLogin: ' + err.message }); }); }
